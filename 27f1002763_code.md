'''python
import os
print(os.listdir("/kaggle/input/competitions/26-t-1-dl-gen-ainppe-1"))

import numpy as np
import pandas as pd
import random
import matplotlib.pyplot as plt
from PIL import Image, ImageEnhance

import tensorflow as tf
from tensorflow.keras.models import Model
from tensorflow.keras.layers import Dense, Dropout, GlobalAveragePooling2D
from tensorflow.keras.applications import EfficientNetB3
from tensorflow.keras.optimizers import Adam

IMAGE_SIZE = 128
BATCH_SIZE = 16
EPOCHS = 10

images_dir = "/kaggle/input/competitions/26-t-1-dl-gen-ainppe-1/images"
train_df =pd.read_csv(os.path.join("/kaggle/input/competitions/26-t-1-dl-gen-ainppe-1/train.csv")) 
test_df = pd.read_csv(os.path.join("/kaggle/input/competitions/26-t-1-dl-gen-ainppe-1/test.csv"))
class_names = train_df.columns[1:]

#loading path
def process_image(path):
    try:
        img = Image.open(path).resize((IMAGE_SIZE, IMAGE_SIZE))
        img = ImageEnhance.Brightness(img).enhance(random.uniform(0.8, 1.2))
        img = ImageEnhance.Contrast(img).enhance(random.uniform(0.8, 1.2))
        return np.array(img) / 255.0
    except:
        return np.zeros((IMAGE_SIZE, IMAGE_SIZE, 3))

def generator(df, images_dir, batch_size):
    while True:
        df=df.sample(frac=1)
        
        for i in range(0, len(df), batch_size):
            batch = df.iloc[i:i+batch_size]
            images=[
                process_image(os.path.join(images_dir, img_id))
                for img_id in batch['id']
            ]
            labels=np.argmax(batch.iloc[:,1:].values, axis=1)

            yield np.array(images), labels

base=EfficientNetB3(
    input_shape=(IMAGE_SIZE, IMAGE_SIZE, 3),
    include_top=False,
    weights='imagenet'
)

for layer in base.layers[:-10]:
    layer.trainable=False

x=GlobalAveragePooling2D()(base.output)
x=Dropout(0.3)(x)
x=Dense(128, activation='relu')(x)
x=Dropout(0.2)(x)

output=Dense(len(class_names), activation='softmax')(x)
model=Model(inputs=base.input, outputs=output)

model.compile(
    optimizer=Adam(1e-4),
    loss='sparse_categorical_crossentropy',
    metrics=['accuracy']
)

#train 
history = model.fit(
    generator(train_df, images_dir, BATCH_SIZE),
    steps_per_epoch=len(train_df)//BATCH_SIZE,
    epochs=EPOCHS
)

#plot
plt.plot(history.history['accuracy'])
plt.plot(history.history['loss'])
plt.legend(['Accuracy', 'Loss'])
plt.show()

test_paths = [
    os.path.join(images_dir, img_id)
    for img_id in test_df['id']
]
    
def load_test(paths):
    return np.array([process_image(p) for p in paths])

preds = model.predict(load_test(test_paths))
preds = np.argmax(preds, axis=1)

#one hot encoding
one_hot_encode=np.zeros((len(preds),len(class_names)))
for i, idx in enumerate(preds):
    one_hot_encode[i, idx]=1
    
submission = pd.DataFrame(one_hot_encode, columns=class_names)
submission.insert(0, "id", test_df['id'])

submission.to_csv("submission.csv", index=False)
print("Submitted")
'''
